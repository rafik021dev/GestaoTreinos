@startuml
class Usuario {
  - int : idUsuario
  - String : nome
  - String : sexo
  - int : idade
  - double : peso
  - double : altura
  - double : metaPeso
  + double : calcularIMC()
  + double : calcularTMB()
  + void : atualizarPeso()
}

class Treino {
  - int : idTreino
  - LocalDate : data
  - TipoTreino : tipo
  - Usuario : usuario
  - List<Exercicio> exercicios
  + void : adicionarExercicio()
}

class Refeição {
  - int : idRefeicao
  - LocalDate : data
  - String : tipoRefeição
  + void : adicionarAlimento()
  + double : calcularCalorias()
}

class Alimento {
  - int : idAlimento
  - String : nomeAlimento
  - double : quantidade
  - double : calorias
  - String : macroNutriente
}

class Sono {
  - int : idSono
  - LocalDate : data
  - double : horasDormidas
  - int : qualidade
}

class Desempenho {
  - int : idDesempenho
  - LocalDate inicioPeriodo
  - LocalDate fimPeriodo
  - double : mediaTreinos
  - double : mediaSono
  - double : mediaCalorias
  - double : indiceDesempenho
  + void : calcularMedias()
}

class Relatorio {
  - int : idRelatorio
  - LocalDate : dataInicio
  - LocalDate : dataFim
  - String : tipo
  + void : gerarRelatorio()
}
class Exercicio{
  - int : idExercicio
  - double : carga
  - int : repeticoes
  - int : series
  - Treino : treino
}

class Alerta {
  - int : idAlerta
  - LocalDate : data
  - String : tipo
  - String : mensagem
}
